  Widget _buildExpenseChart(AppProvider appProvider) {
    final categoryExpenses = appProvider.categoryExpenses;
    
    if (categoryExpenses.isEmpty) {
      return Card(
        child: Container(
          height: 200,
          padding: const EdgeInsets.all(16),
          child: const Center(
            child: Text('No hay gastos este mes'),
          ),
        ),
      );
    }
    
    // Preparar datos para el gr√°fico
    List<PieChartSectionData> sections = [];
    List<Color> colors = [
      AppTheme.foodColor,
      AppTheme.transportColor,
      AppTheme.entertainmentColor,
      AppTheme.shoppingColor,
      Colors.teal,
      Colors.purple,
      Colors.orange,
      Colors.pink,
      Colors.indigo,
    ];
    
    int colorIndex = 0;
    categoryExpenses.forEach((category, amount) {
      if (amount > 0) {
        final percentage = appProvider.getCategoryPercentage(category);
        sections.add(
          PieChartSectionData(
            value: amount,
            title: '${percentage.toStringAsFixed(0)}%',
            color: colors[colorIndex % colors.length],
            radius: 50,
            titleStyle: const TextStyle(
              fontSize: 12,
              fontWeight: FontWeight.bold,
              color: Colors.white,
            ),
          ),
        );
        colorIndex++;
      }
    });
    
    return Card(
      child: Container(
        height: 250,
        padding: const EdgeInsets.all(16),
        child: Column(
          children: [
            Expanded(
              child: PieChart(
                PieChartData(
                  sections: sections,
                  sectionsSpace: 2,
                  centerSpaceRadius: 40,
                ),
              ),
            ),
            const SizedBox(height: 16),
            // Leyenda
            Wrap(
              spacing: 16,
              runSpacing: 8,
              children: categoryExpenses.entries.map((entry) {
                if (entry.value <= 0) return const SizedBox.shrink();
                final transaction = Transaction(
                  title: '',
                  amount: 0,
                  type: TransactionType.expense,
                  category: entry.key,
                  date: DateTime.now(),
                );
                return Row(
                  mainAxisSize: MainAxisSize.min,
                  children: [
                    Container(
                      width: 12,
                      height: 12,
                      decoration: BoxDecoration(
                        color: colors[categoryExpenses.keys.toList().indexOf(entry.key) % colors.length],
                        shape: BoxShape.circle,
                      ),
                    ),
                    const SizedBox(width: 4),
                    Text(
                      '${transaction.getCategoryIcon()} ${transaction.getCategoryName()}',
                      style: const TextStyle(fontSize: 12),
                    ),
                  ],
                );
              }).toList(),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildQuickActions() {
    return GridView.count(
      shrinkWrap: true,
      physics: const NeverScrollableScrollPhysics(),
      crossAxisCount: 4,
      children: [
        _quickActionButton(
          Icons.add_circle_outline,
          'Ingreso',
          Colors.green,
          () => _showQuickAddDialog(TransactionType.income),
        ),
        _quickActionButton(
          Icons.remove_circle_outline,
          'Gasto',
          Colors.red,
          () => _showQuickAddDialog(TransactionType.expense),
        ),
        _quickActionButton(
          Icons.credit_card,
          'Deuda',
          Colors.orange,
          () => _showAddDebtDialog(),
        ),
        _quickActionButton(
          Icons.flag,
          'Meta',
          Colors.blue,
          () => _showAddGoalDialog(),
        ),
      ],
    );
  }

  Widget _quickActionButton(IconData icon, String label, Color color, VoidCallback onTap) {
    return InkWell(
      onTap: onTap,
      borderRadius: BorderRadius.circular(12),
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Container(
            padding: const EdgeInsets.all(12),
            decoration: BoxDecoration(
              color: color.withOpacity(0.1),
              borderRadius: BorderRadius.circular(12),
            ),
            child: Icon(icon, color: color),
          ),
          const SizedBox(height: 4),
          Text(
            label,
            style: const TextStyle(fontSize: 12),
          ),
        ],
      ),
    );
  }

  Widget _buildGoalsSection(AppProvider appProvider) {
    return Column(
      children: appProvider.activeGoals.take(3).map((goal) {
        return _goalCard(goal, appProvider);
      }).toList(),
    );
  }

  Widget _goalCard(PersonalGoal goal, AppProvider appProvider) {
    return Card(
      margin: const EdgeInsets.only(bottom: 12),
      child: InkWell(
        onTap: () => _showGoalDetailsDialog(goal),
        borderRadius: BorderRadius.circular(12),
        child: Padding(
          padding: const EdgeInsets.all(16),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Row(
                children: [
                  Text(
                    goal.icon ?? 'üéØ',
                    style: const TextStyle(fontSize: 24),
                  ),
                  const SizedBox(width: 8),
                  Expanded(
                    child: Column(
                      crossAxisAlignment: CrossAxisAlignment.start,
                      children: [
                        Text(
                          goal.title,
                          style: const TextStyle(
                            fontWeight: FontWeight.w600,
                            fontSize: 16,
                          ),
                        ),
                        if (goal.daysRemaining > 0)
                          Text(
                            '${goal.daysRemaining} d√≠as restantes',
                            style: TextStyle(
                              color: Colors.grey[600],
                              fontSize: 12,
                            ),
                          ),
                      ],
                    ),
                  ),
                  Text(
                    '${(goal.progress * 100).toInt()}%',
                    style: TextStyle(
                      color: Color(goal.color),
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                ],
              ),
              const SizedBox(height: 12),
              LinearProgressIndicator(
                value: goal.progress,
                backgroundColor: Colors.grey[200],
                valueColor: AlwaysStoppedAnimation<Color>(
                  Color(goal.color),
                ),
              ),
              const SizedBox(height: 8),
              Text(
                '${_currencyFormat.format(goal.currentAmount)} de ${_currencyFormat.format(goal.targetAmount)}',
                style: TextStyle(
                  color: Colors.grey[600],
                  fontSize: 12,
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }

  Widget _debtCard(Debt debt, AppProvider appProvider) {
    return Card(
      margin: const EdgeInsets.only(bottom: 12),
      child: InkWell(
        onTap: () => _showDebtDetailsDialog(debt),
        borderRadius: BorderRadius.circular(12),
        child: Padding(
          padding: const EdgeInsets.all(16),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Row(
                children: [
                  Text(
                    debt.type.icon,
                    style: const TextStyle(fontSize: 24),
                  ),
                  const SizedBox(width: 8),
                  Expanded(
                    child: Column(
                      crossAxisAlignment: CrossAxisAlignment.start,
                      children: [
                        Text(
                          debt.title,
                          style: const TextStyle(
                            fontWeight: FontWeight.w600,
                            fontSize: 16,
                          ),
                        ),
                        if (debt.creditor != null)
                          Text(
                            debt.creditor!,
                            style: TextStyle(
                              color: Colors.grey[600],
                              fontSize: 12,
                            ),
                          ),
                      ],
                    ),
                  ),
                  Column(
                    crossAxisAlignment: CrossAxisAlignment.end,
                    children: [
                      Text(
                        _currencyFormat.format(debt.remainingAmount),
                        style: const TextStyle(
                          color: Colors.orange,
                          fontWeight: FontWeight.bold,
                          fontSize: 16,
                        ),
                      ),
                      Text(
                        'Pago: ${_currencyFormat.format(debt.minimumPayment)}',
                        style: TextStyle(
                          color: Colors.grey[600],
                          fontSize: 10,
                        ),
                      ),
                    ],
                  ),
                ],
              ),
              const SizedBox(height: 12),
              LinearProgressIndicator(
                value: debt.progress,
                backgroundColor: Colors.grey[200],
                valueColor: const AlwaysStoppedAnimation<Color>(
                  Colors.orange,
                ),
              ),
              const SizedBox(height: 8),
              Row(
                mainAxisAlignment: MainAxisAlignment.spaceBetween,
                children: [
                  Text(
                    '${(debt.progress * 100).toInt()}% pagado',
                    style: TextStyle(
                      color: Colors.grey[600],
                      fontSize: 12,
                    ),
                  ),
                  if (debt.nextPaymentDate != null)
                    Text(
                      'Pr√≥ximo pago: ${DateFormat('dd/MM').format(debt.nextPaymentDate!)}',
                      style: TextStyle(
                        color: Colors.grey[600],
                        fontSize: 12,
                      ),
                    ),
                ],
              ),
            ],
          ),
        ),
      ),
    );
  }
